select_one: 
    move edx, 1 
    mov cl, al 
    shl edx, cl 
    not edx 
    and ebx, edx 
    ret 


swap: 
    push edx
    mov eax, edx 
    shl eax, 16
    shr edx, 16 
    mov ax, dx 
    pop edx
    ret 


copy_ebx: 
    push ebx 
    and ebx, 0x000007C0
    and eax, 0xFFFFF83F
    or eax, ebx
    pop ebx 
    ret 


first_bits_of_ebx: 
    mov edx, 0xFFFFFFFF
    mov cl, al 
    shl edx, cl 
    not edx
    or ebx, edx
