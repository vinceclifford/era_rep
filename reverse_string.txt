palindrom_test: 
    push ebx
    push edx 
    mov ebx, eax 

beginning_first_loop: 
    cmp BYTE [ebx], 0 
    je pointers_are_set 
    inc ebx 
    jmp beginning__first_loop

pointers_are_set: 
    dec ebx

beginning_second_loop: 
    cmp ebx, eax
    jle endresult 
    mov edx, 0 
    mov dl, [eax]
    xchg dl, [ebx]
    mov [eax], dl
    inc eax 
    dec ebx 
    jump beginning_second_loop

endresult: 
    pop edx 
    pop ebx 
    ret
